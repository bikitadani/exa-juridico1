Tabela de Sociedades CREATE TABLE IF NOT EXISTS sociedades ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, nome TEXT NOT NULL, cnpj TEXT, cor_index INTEGER DEFAULT 0, atos TEXT[] DEFAULT '{}', created_at TIMESTAMPTZ DEFAULT now(), updated_at TIMESTAMPTZ DEFAULT now() ); -- 2. Tabela de Processos (cards do Kanban) CREATE TABLE IF NOT EXISTS processos ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, sociedade TEXT NOT NULL, tipo_ato TEXT NOT NULL, responsavel TEXT DEFAULT '', etapa INTEGER DEFAULT 1, observacoes TEXT DEFAULT '', cor_index INTEGER DEFAULT 0, -- DocuSign docusign_link TEXT DEFAULT '', docusign_status TEXT DEFAULT '—', docusign_signatarios TEXT DEFAULT '', -- Protocolo / Junta numero_protocolo TEXT DEFAULT '', data_protocolo DATE, prazo_previsto DATE, -- Exigência exigencia TEXT DEFAULT '', data_exigencia DATE, -- Registro data_registro DATE, -- NF / Pagamento nf_numero TEXT DEFAULT '', nf_valor TEXT DEFAULT '', nf_data DATE, data_pagamento DATE, -- Anexos (JSON com as 9 categorias) anexos JSONB DEFAULT '{ "ato_elaborado": null, "ato_assinado": null, "comprovante_protocolo": null, "exigencia_apontada": null, "ato_exigencia_cumprida": null, "novo_protocolo": null, "ato_registrado": null, "despesa": null, "comprovante_pagamento": null }'::jsonb, -- Arquivo de rodadas anteriores anexos_arquivados JSONB DEFAULT '[]'::jsonb, created_at TIMESTAMPTZ DEFAULT now(), updated_at TIMESTAMPTZ DEFAULT now() ); -- 3. Tabela de Histórico de Movimentações CREATE TABLE IF NOT EXISTS historicos ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, processo_id UUID REFERENCES processos(id) ON DELETE CASCADE, etapa INTEGER NOT NULL, descricao TEXT DEFAULT '', responsavel TEXT DEFAULT '', data TIMESTAMPTZ DEFAULT now() ); -- 4. Índices para performance CREATE INDEX IF NOT EXISTS idx_processos_etapa ON processos(etapa); CREATE INDEX IF NOT EXISTS idx_processos_sociedade ON processos(sociedade); CREATE INDEX IF NOT EXISTS idx_historicos_processo ON historicos(processo_id); -- 5. Row Level Security (RLS) — Permitir acesso público (anon key) -- Em produção, você pode adicionar autenticação depois ALTER TABLE sociedades ENABLE ROW LEVEL SECURITY; ALTER TABLE processos ENABLE ROW LEVEL SECURITY; ALTER TABLE historicos ENABLE ROW LEVEL SECURITY; -- Policies para permitir todas as operações com a anon key CREATE POLICY "Allow all on sociedades" ON sociedades FOR ALL USING (true) WITH CHECK (true); CREATE POLICY "Allow all on processos" ON processos FOR ALL USING (true) WITH CHECK (true); CREATE POLICY "Allow all on historicos" ON historicos FOR ALL USING (true) WITH CHECK (true); -- 6. Função para atualizar updated_at automaticamente CREATE OR REPLACE FUNCTION update_updated_at() RETURNS TRIGGER AS $$ BEGIN NEW.updated_at = now(); RETURN NEW; END; $$ LANGUAGE plpgsql; CREATE TRIGGER tr_processos_updated BEFORE UPDATE ON processos FOR EACH ROW EXECUTE FUNCTION update_updated_at(); CREATE TRIGGER tr_sociedades_updated BEFORE UPDATE ON sociedades FOR EACH ROW EXECUTE FUNCTION update_updated_at();
